001    package com.fairportfirst.frc2014;
002    
003    import com.fairportfirst.frc2014.commands.BoomExtendCommand;
004    import com.fairportfirst.frc2014.commands.BoomRetractCommand;
005    import com.fairportfirst.frc2014.commands.BoomSpinCommand;
006    import com.fairportfirst.frc2014.commands.BoomSpinReverseCommand;
007    import com.fairportfirst.frc2014.commands.DisableLightCommand;
008    import com.fairportfirst.frc2014.commands.EnableLightsCommand;
009    import com.fairportfirst.frc2014.commands.ExtendShooterCommand;
010    import com.fairportfirst.frc2014.commands.RetractShooterCommand;
011    import com.fairportfirst.frc2014.commands.ShootCommand;
012    import com.fairportfirst.frc2014.commands.ToggleBoomCommand;
013    import com.fairportfirst.frc2014.commands.ToggleHighSpeedCommand;
014    import edu.wpi.first.wpilibj.Joystick;
015    import edu.wpi.first.wpilibj.buttons.Button;
016    import edu.wpi.first.wpilibj.buttons.JoystickButton;
017    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
018    
019    /**
020     * This class is the glue that binds the controls on the physical operator
021     * interface to the commands and command groups that allow control of the robot.
022     */
023    public class OI {
024        //// CREATING BUTTONS
025        // One type of button is a joystick button which is any button on a joystick.
026        // You create one by telling it which joystick it's on and which button
027        // number it is.
028        // Joystick stick = new Joystick(port);
029        // Button button = new JoystickButton(stick, buttonNumber);
030    
031        // Another type of button you can create is a DigitalIOButton, which is
032        // a button or switch hooked up to the cypress module. These are useful if
033        // you want to build a customized operator interface.
034        // Button button = new DigitalIOButton(1);
035        // There are a few additional built in buttons you can use. Additionally,
036        // by subclassing Button you can create custom triggers and bind those to
037        // commands the same as any other Button.
038        //// TRIGGERING COMMANDS WITH BUTTONS
039        // Once you have a button, it's trivial to bind it to a button in one of
040        // three ways:
041        // Start the command when the button is pressed and let it run the command
042        // until it is finished as determined by it's isFinished method.
043        // button.whenPressed(new ExampleCommand());
044        // Run the command while the button is being held down and interrupt it once
045        // the button is released.
046        // button.whileHeld(new ExampleCommand());
047        // Start the command when the button is released  and let it run the command
048        // until it is finished as determined by it's isFinished method.
049        // button.whenReleased(new ExampleCommand());
050        private Joystick leftJoy;
051        private Joystick rightJoy;
052    
053        private Joystick gamePad;
054    
055        private Button boomExtendButton;
056        private Button boomRetractButton;
057        private Button boomSpinButton;
058        private Button boomSpinReverseButton;
059        private Button shootButton;
060        private Button shiftCommand;
061        private Button turnOnLightsButton;
062        private Button turnOffLightsButton;
063    
064        public OI() {
065            
066            leftJoy = new Joystick(RobotMap.DRIVER_LEFT_JOY_ID);
067            rightJoy = new Joystick(RobotMap.DRIVER_RIGHT_JOY_ID);
068            gamePad = new Joystick(RobotMap.OPERATOR_CONTROLLER_ID);
069    
070            //TODO: FIX BUTTON IDs
071            shootButton = new JoystickButton(leftJoy, RobotMap.OPERATOR_SHOOTER_BUTTON);
072            shootButton.whenPressed(new ShootCommand());
073            boomExtendButton = new JoystickButton(rightJoy, RobotMap.OPERATOR_BOOM_BUTTON);
074            SmartDashboard.getBoolean("Operator Right Handed", true);
075            boomExtendButton.whenPressed(new BoomExtendCommand());
076            boomRetractButton = new JoystickButton(leftJoy, RobotMap.OPERATOR_BOOM_BUTTON);
077            boomRetractButton.whenPressed(new BoomRetractCommand());
078            boomSpinButton = new JoystickButton(leftJoy, RobotMap.OPERATOR_FEEDER_FORWARD_BUTTON);
079            boomSpinButton.whileHeld(new BoomSpinCommand());
080            boomSpinReverseButton = new JoystickButton(leftJoy, RobotMap.OPERATOR_FEEDER_REVERSE_BUTTON);
081            boomSpinReverseButton.whileHeld(new BoomSpinReverseCommand());
082            shiftCommand = new JoystickButton(leftJoy, 5);
083            shiftCommand.whenPressed(new ToggleHighSpeedCommand());
084      
085            turnOnLightsButton = new JoystickButton(rightJoy, 11);
086            turnOnLightsButton.whenPressed(new EnableLightsCommand());
087            turnOffLightsButton = new JoystickButton(leftJoy, 11);
088            turnOffLightsButton.whenPressed(new DisableLightCommand());
089            
090        }
091    
092        public double getLeftStickY() {
093            return leftJoy.getY();
094        }
095    
096        public double getRightStickY() {
097            return rightJoy.getY();
098        }
099    
100        public double getYAxis() {
101            return leftJoy.getY();
102        }
103    
104        public double getXAxis() {
105            return leftJoy.getX();
106        }
107    
108        public double getZAxis() {
109            return leftJoy.getZ();
110        }
111    
112        public boolean isBoomSpinButtonDown() {
113            return boomSpinButton.get();
114        }
115    
116        public boolean isBoomSpinReverseButtonDown() {
117            return boomSpinReverseButton.get();
118        }
119        
120        public boolean isEnableLightsButtonDown() {
121            return turnOnLightsButton.get();
122        }
123    
124    }
