005    package com.fairportfirst.frc2014.subsystems;
006    
007    import com.fairportfirst.frc2014.RobotMap;
008    import edu.wpi.first.wpilibj.Solenoid;
009    import edu.wpi.first.wpilibj.command.Subsystem;
010    
011    /**
012     *
013     * @author Tyler
014     */
015    public class ShooterSubsystem extends Subsystem {
016        // Put methods for controlling this subsystem
017        // here. Call these from Commands.
018     
019        private static ShooterSubsystem instance;
020        
021        Solenoid shooter1Extend;
022        Solenoid shooter1Retract;
023        Solenoid shooter2Extend;
024        Solenoid shooter2Retract;
025        Solenoid shooter3Extend;
026        Solenoid shooter3Retract;
027        Solenoid shooter4Extend;
028        Solenoid shooter4Retract;
029    
030        public ShooterSubsystem() {
031            shooter1Extend = new Solenoid(RobotMap.SHOOTER1_E_ID);
032            shooter1Retract = new Solenoid(RobotMap.SHOOTER1_R_ID);
033            shooter2Extend = new Solenoid(RobotMap.SHOOTER2_E_ID);
034            shooter2Retract = new Solenoid(RobotMap.SHOOTER2_R_ID);
035            shooter3Extend = new Solenoid(RobotMap.SHOOTER3_E_ID);
036            shooter3Retract = new Solenoid(RobotMap.SHOOTER3_R_ID);
037            shooter4Extend = new Solenoid(RobotMap.SHOOTER4_E_ID);
038            shooter4Retract = new Solenoid(RobotMap.SHOOTER4_R_ID);
039        }
040        
041        public void initDefaultCommand() {
042            // Set the default command for a subsystem here.
043            //setDefaultCommand(new MySpecialCommand());
044        }
045        
046        public void extend(){
047            shooter1Retract.set(false);
048            shooter2Retract.set(false);
049            shooter3Retract.set(false);
050            shooter4Retract.set(false);
051            
052            shooter1Extend.set(true);
053            shooter2Extend.set(true);
054            shooter3Extend.set(true);
055            shooter4Extend.set(true);
056        }
057        
058        public void retract(){
059            shooter1Extend.set(false);
060            shooter2Extend.set(false);
061            shooter3Extend.set(false);
062            shooter4Extend.set(false);
063            
064            shooter1Retract.set(true);
065            shooter2Retract.set(true);
066            shooter3Retract.set(true);
067            shooter4Retract.set(true);
068        }
069        
070        public boolean isExtended(){
071            return shooter1Extend.get();
072        }
073        
074        public static ShooterSubsystem getInstance(){
075            
076            if(instance == null){
077                instance = new ShooterSubsystem();
078            }
079            
080            return instance;
081        }
082        
083    }
